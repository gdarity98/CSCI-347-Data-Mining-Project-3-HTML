10   def k_means(data_matrix, k, epsilon):
11       # Randomly select Data points to be the means
12       #   I'm not sure if this is bad to do,
13       #   should I be generating random number within the range of the data set?
14       #   If I dont hard code the means I get different answers form the test cases
15       # the index of the means is going to be the cluster assigned.
16       max_random = len(data_matrix)
17       index = random.sample(range(0, max_random), k)
18       # randomly initializing the means from the data set
19       means = np.zeros((k, len(data_matrix[0])))
20       for i in range(k):
21           d_instance = data_matrix[index[i]]
22           means[i] = d_instance
23   
24       # hard coding what the means are
25       # Test Case 1

27       # Test Case 2
28       # means = np.array([[3.67371566, 3.05039513], [-2.27144128, 0.58447117], [4.50290941, 4.78503096]])
29       # Test Case 3
30       # means = np.array([[3.75224869, -2.9680029], [-1.45630451, -1.65071389], [-3.16968936, -1.47368135]])
31   
32       new_means = np.zeros((k, len(data_matrix[0])))
33       # print(means)
34       # print(new_means)
35       run = True
36       while run:
37           # This is what keeps track of what cluster a data instance has been assigned to.
38           cluster_assignment = np.zeros(len(data_matrix))
39           # print(len(cluster_assignment))
40   
41           # Now I need to assign each point to the cluster with the closest mean
42           # if there are ties then pick the mean with the lowest index
43           for data in range(len(data_matrix)):
44               distance = np.zeros(len(means))
45               for i in range(len(means)):
46                   for j in range(len(means[i])):
47                       x_i = data_matrix[data][j]
48                       mu_i = means[i][j]
49                       distance[i] += (data_matrix[data][j] - means[i][j]) ** 2
50               # Numpy.argmin() selects the indices corresponding to the first occurrence if there is a tie
51               # therefore if the distances look like [0, 5, 0] then the point will be assigned to cluster 0
52               cluster = np.argmin(distance)
53               cluster_assignment[data] = cluster + 1
54   
55           # recompute the means based on cluster membership
56           for mean in range(len(new_means)):
57               count = 0
58               for data in range(len(data_matrix)):
59                   if cluster_assignment[data] == mean + 1:
60                       for i in range(len(new_means[0])):
61                           new_means[mean][i] += data_matrix[data][i]
62                       count += 1
63               if count != 0:
64                   for i in range(len(new_means[0])):
65                       new_means[mean][i] = new_means[mean][i] / count
66               if count == 0:
67                   # print("No Attributes Assigned to cluster:", mean)
68                   for i in range(len(new_means[0])):
69                       new_means[mean][i] = None
70   
71           # keep going until epsilon
72   
73           for i in range(len(means)):
74               for j in range(len(means[0])):
75                   x = means[i][j]
76                   y = new_means[i][j]
77                   if abs(means[i][j] - new_means[i][j]) <= epsilon:
78                       run = False
79           else:
80               means = np.copy(new_means)
81               new_means = np.zeros((k, len(data_matrix[0])))
82   
83       # print(cluster_assignment)
84       # print(means)
85       return cluster_assignment, means

107  def dbscan(data_matrix, minpts, epsilon):
108      # This is what keeps track of what cluster a data instance has been assigned to.
109      cluster_assignment = np.zeros(len(data_matrix))
110      # print(len(cluster_assignment))
111  
112      core = np.zeros(len(data_matrix))
113      border = np.zeros(len(data_matrix))
114      noise = np.zeros(len(data_matrix))
115      epsilon_neighborhood = np.zeros((len(data_matrix), len(data_matrix)))
116      # print(data_matrix[0])
117  
118      # find all the core points
119      # compute the epsilon neighborhood of x_i
120      # find the euclidean distance between this point and every other
121      # and if it is <= epsilon add it to neighborhood
122      # use np.append(data, axis=0)
123      # initialized the cluster label of x_i to be 0
124      # if the size of the epsilon neighborhood is >= minpts then
125      # add x_i to core
126      # else
127      # add x_i to border
128      for data in range(len(data_matrix)):
129          neighborhood = np.array([])
130          for possible_neighbor in range(len(data_matrix)):
131              if possible_neighbor != data:
132                  distance = 0
133                  for i in range(len(data_matrix[data])):
134                      point_1 = data_matrix[data][i]
135                      point_2 = data_matrix[possible_neighbor][i]
136                      distance += (data_matrix[possible_neighbor][i] - data_matrix[data][i]) ** 2
137                  distance = math.sqrt(distance)
138                  if distance < epsilon:
139                      neighborhood = np.append(neighborhood, possible_neighbor)
140          if len(neighborhood) != 0:
141              for i in neighborhood:
142                  index = int(i)
143                  epsilon_neighborhood[data][index] = 1
144              if len(neighborhood) >= minpts:
145                  core[data] = 1
146              # elif len(epsilon_neighborhood) > 0:
147              #     border[data] = 1
148  
149      # now we are going to go through all the core points
150      # grow out the cluster
151      k = 0
152      for data in range(len(core)):
153          # check and make sure data has no label
154          if cluster_assignment[data] == 0 and core[data] == 1:
155              k += 1
156              cluster_assignment[data] = k
157              # DensityConnected is a function that lets us explore the epsilon neighborhood of x_i
158              # and add all the points in the neighborhood to the same cluster as x_i
159              # as well as any core points with x_i neighborhood and their epsilon neighborhood points
160              before_core = np.copy(core)
161              cluster_assignment = density_connected(data, k, epsilon_neighborhood, cluster_assignment, core)
162              core = np.copy(before_core)
163  
164      # If there is a point in cluster_assignment the has a label of 0
165      # add it to noise
166      for i in range(len(cluster_assignment)):
167          if cluster_assignment[i] == 0:
168              noise[i] = 1
169  
170      for i in range(len(core)):
171          if core[i] == 0 and noise[i] == 0:
172              border[i] = 1
173  
174      # print("CA:", cluster_assignment, "\n", "CORE:",  core, "\n", "BORDER:",  border, "\n", "NOISE:", noise)
175      return cluster_assignment, core, border, noise

178  def density_connected(x, k, en, ca, core):
179      # for each data in en[x]
180      # ca[data] = k
181      # if data in core
182      # ca = density_connected(data, k, en, ca, core)
183      for index in range(len(en[x])):
184          if en[x][index] == 1:
185              ca[index] = k
186              if core[index] == 1:
187                  # added core[x] = 0 to stop inf recursion
188                  # if a core point is going to be checked then no longer consider it a core point
189                  core[x] = 0
190                  ca = density_connected(index, k, en, ca, core)
191      return ca
